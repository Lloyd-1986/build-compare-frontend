import express from "express"; import cors from "cors"; import projectRoutes from "./routes/projects.js"; import materialRoutes from "./routes/materials.js"; import priceRoutes from "./routes/prices.js"; import billingRoutes from "./routes/billing.js"; import webhookRoutes from "./routes/webhook.js"; const app = express(); app.use(cors()); app.use(express.json()); app.use("/api/projects", projectRoutes); app.use("/api/materials", materialRoutes); app.use("/api/prices", priceRoutes); app.use("/api/billing", billingRoutes); app.use("/api/webhook", webhookRoutes); app.get("/health", (_, res) => res.send("OK")); app.listen(process.env.PORT || 3000);
